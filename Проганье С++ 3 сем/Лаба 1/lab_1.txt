#include <iostream>
#include <math.h>
#include <cmath>

using namespace std;

int main()
{
	setlocale(LC_ALL, "Russian");
	//Обьявление входных данных.
	int b = -3, m = 2;
	float x, y, z, a = -1.1, k = 0.1;

	x = (exp(-2 * b) + pow((abs(pow(a, 3))), (1 / m)))/((pow(b, 4))*(1.5+((a-b)/(a+b))));

	y = (((a*b)/k) + (m/(a*b)))*(log(abs(x + a)));
	
	z = m*(tan(pow((x - 5), 2)))*(acos(k*y));
	
	//Вывод значений функций
	cout << "Вычисление математических функций." << endl;
	cout << "Результаты вычислений: " << endl;
	cout << "x= " << x << endl;
	cout << "y= " << y << endl;
	cout << "z=: " << z << endl;
	system("Pause");
	return 0;
}
